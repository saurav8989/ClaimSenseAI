  initDb();
  const data = fs.readFileSync(dbPath, 'utf8');
  return JSON.parse(data);

export function getClaimById(claimId) {
  const claims = getClaims();
  return claims.find(c => c.claimId === claimId) || null;
}

export function saveClaim(claim) {
  initDb();
  const claims = getClaims();
  claims.push(claim);
  fs.writeFileSync(dbPath, JSON.stringify(claims, null, 2), 'utf8');
  return claim;
}

export function updateClaimStatus(claimId, status, comments = "") {
  initDb();
  const claims = getClaims();
  const idx = claims.findIndex(c => c.claimId === claimId);
  if (idx !== -1) {
    claims[idx].status = status;
    claims[idx].reviewerComments = comments;
    claims[idx].reviewedAt = new Date().toISOString();
    fs.writeFileSync(dbPath, JSON.stringify(claims, null, 2), 'utf8');
    return claims[idx];
  }
  return null;
}

import { getClaims, getClaimById, saveClaim, updateClaimStatus } from '@/lib/mockDb';

// Attempt to import Dev 2's engines. Fallback gracefully if not yet created.
let validateClinicalServices = null;
let evaluateStpCompliance = null;
let calculateRiskScore = null;

try {
  const clinicalValidator = require('@/lib/clinicalValidator');
  validateClinicalServices = clinicalValidator.validateClinicalServices;
} catch (e) {}

try {
  const stpEngine = require('@/lib/stpEngine');
  evaluateStpCompliance = stpEngine.evaluateStpCompliance;
} catch (e) {}

try {
  const riskScorer = require('@/lib/riskScorer');
  calculateRiskScore = riskScorer.calculateRiskScore;
} catch (e) {}

export async function GET(request) {
  const { searchParams } = new URL(request.url);
  const claimId = searchParams.get('claimId');

  if (claimId) {
    const claim = getClaimById(claimId);
    if (!claim) return Response.json({ error: "Claim not found" }, { status: 404 });
    return Response.json(claim);
  }

  const claims = getClaims();
  return Response.json(claims);
}

export async function POST(request) {
  try {
    const claimPayload = await request.json();
    const claimId = `CLM-${Math.floor(10000 + Math.random() * 90000)}`;
    const submittedAt = new Date().toISOString();

    // 1. Prepare base claim
    const newClaim = {
      ...claimPayload,
      claimId,
      submittedAt,
      status: "PENDING_REVIEW"
    };

    // 2. Run Analysis Engines if available, else fall back to static/stub assessment
    let clinicalValidation = { isValid: true, issues: [] };
    let stpCompliance = { isCompliant: true, complianceScore: 100, protocolName: "General Consultation Protocol", deviations: [] };
    let riskScoring = { overallRiskScore: 0, riskCategory: "LOW", riskFlags: [], reviewerPriority: 5 };

    if (validateClinicalServices && evaluateStpCompliance && calculateRiskScore) {
      clinicalValidation = validateClinicalServices(newClaim);
      stpCompliance = evaluateStpCompliance(newClaim);
      riskScoring = calculateRiskScore(clinicalValidation, stpCompliance, newClaim);
    } else {
      // Mock Fallback Logic for early testing before Dev 2 is finished
      const primaryDiag = newClaim.diagnoses.find(d => d.isPrimary) || newClaim.diagnoses[0];
      const code = primaryDiag?.code || "UNKNOWN";
      
      // Stub check: if Malaria code is selected and no test is present
      if (code.startsWith("1F4")) {
        const hasTest = newClaim.carePathway.some(step => step.code === "TEST-MAL-RDT" || step.code === "TEST-MAL-MIC");
        if (!hasTest) {
          stpCompliance = {
            isCompliant: false,
            complianceScore: 60,
            protocolName: "Malaria Standard Treatment Protocol",
            deviations: [{ type: "TEST_REQUIRED", message: "Prescribing antimalarials requires a prior diagnostic verification (RDT or Microscopy).", penalty: 40 }]
          };
          riskScoring = { overallRiskScore: 40, riskCategory: "MEDIUM", riskFlags: ["HIGH_PROTOCOL_DEVIATION"], reviewerPriority: 3 };
        }
      }
    }

    newClaim.clinicalValidation = clinicalValidation;
    newClaim.stpCompliance = stpCompliance;
    newClaim.riskScoring = riskScoring;

    // Automatically auto-approve low-risk claims (Score 0-30)
    if (riskScoring.overallRiskScore <= 30) {
      newClaim.status = "APPROVED";
    }

    saveClaim(newClaim);

    return Response.json(newClaim);
  } catch (error) {
    console.error("Error creating claim:", error);
    return Response.json({ error: "Internal Server Error", details: error.message }, { status: 500 });
  }
}

export async function PUT(request) {
  try {
    const { claimId, status, comments } = await request.json();
    if (!claimId || !status) {
      return Response.json({ error: "Missing claimId or status" }, { status: 400 });
    }

    const updated = updateClaimStatus(claimId, status, comments);
    if (!updated) return Response.json({ error: "Claim not found" }, { status: 404 });

    return Response.json(updated);
  } catch (error) {
    console.error("Error updating claim:", error);
    return Response.json({ error: "Internal Server Error" }, { status: 500 });
  }
}

  const handleAddDiagnosis = (suggestion) => {
    // Avoid duplicates
    if (diagnoses.some(d => d.code === suggestion.code)) return;
    
    const isFirst = diagnoses.length === 0;
    const newDiag = {
      code: suggestion.code,
      name: suggestion.title,
      isPrimary: isFirst
    };
    
    setDiagnoses([...diagnoses, newDiag]);
  };

  // Submit claim handler
  const handleSubmitClaim = async (payload) => {
    setIsSubmitting(true);
    setAnalysisResult(null);
    try {
      const res = await fetch('/api/openimis/claims', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
      const data = await res.json();
      setAnalysisResult(data);
      
      // If success, clear form fields
      if (res.ok) {
        setCarePathway([]);
        setDiagnoses([]);
        setNoteText('');
        setSuggestions([]);
      }
    } catch (error) {
      alert("Failed to submit claim.");
      console.error(error);
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <div className="min-h-screen bg-slate-50 dark:bg-black font-sans py-10 px-6 sm:px-12">
      <div className="max-w-7xl mx-auto space-y-8">
        
        {/* Header Section */}
        <div className="flex justify-between items-center bg-white/40 dark:bg-zinc-900/40 backdrop-blur border border-slate-200 dark:border-zinc-800 p-6 rounded-2xl">
          <div>
            <h1 className="text-2xl font-black text-slate-800 dark:text-zinc-50 tracking-tight">
              👨‍⚕️ Provider Workspace
            </h1>
            <p className="text-sm text-slate-500 dark:text-zinc-400">
              Prepare, validate, and submit clinical claims to openIMIS.
            </p>
          </div>
          <div className="text-right text-xs bg-teal-100 dark:bg-teal-950/40 border border-teal-200 dark:border-teal-900 text-teal-800 dark:text-teal-200 px-3 py-1.5 rounded-xl font-bold">
            👤 Dr. Sarah Jenkins (PROV-9082)
          </div>
        </div>

        {/* Dashboard Grid */}
        <div className="grid grid-cols-1 lg:grid-cols-12 gap-8 items-start">
          
          {/* LEFT: Clinical Notes & Suggestions (5 cols) */}
          <div className="lg:col-span-5 space-y-6">
            
            <div className="bg-white/80 dark:bg-zinc-900/80 backdrop-blur-md border border-slate-200 dark:border-zinc-800 rounded-2xl p-6 shadow-xl space-y-4">
              <h2 className="text-lg font-bold text-slate-800 dark:text-zinc-200">
                📝 Clinical Notes
              </h2>
              
              <textarea
                value={noteText}
                onChange={e => setNoteText(e.target.value)}
                placeholder="Write or paste your patient summary notes here... (e.g., Patient presented with fever, chills, and muscle aches for three days. Suspect malaria.)"
                rows={6}
                className="w-full px-4 py-3 border rounded-xl dark:bg-zinc-800 dark:border-zinc-700 text-sm focus:outline-none focus:ring-2 focus:ring-teal-500"
              />
              
              <button
                onClick={handleSuggestCodes}
                disabled={isSearching || !noteText.trim()}
                className="w-full py-2.5 bg-teal-600 hover:bg-teal-700 text-white rounded-xl text-sm font-bold shadow-md hover:shadow-lg disabled:opacity-50 transition-all cursor-pointer"
              >
                {isSearching ? "Analyzing symptoms..." : "✨ Suggest ICD Diagnoses"}
              </button>
            </div>

            {/* ICD Suggestions Output */}
            <div className="bg-white/80 dark:bg-zinc-900/80 backdrop-blur-md border border-slate-200 dark:border-zinc-800 rounded-2xl p-6 shadow-xl space-y-4">
              <h3 className="text-xs font-bold text-slate-500 dark:text-zinc-400 uppercase tracking-wider">
                Recommended Diagnoses
              </h3>
              
              {suggestions.length === 0 ? (
                <p className="text-xs text-slate-400 italic">No recommendations yet. Run note analysis above.</p>
              ) : (
                <div className="space-y-3">
                  {suggestions.map((item, idx) => (
                    <div 
                      key={idx} 
                      onClick={() => handleAddDiagnosis(item)}
                      className="group border border-slate-100 dark:border-zinc-800 bg-slate-50/50 dark:bg-zinc-950/20 p-3 rounded-xl hover:border-teal-500 hover:bg-teal-50/10 transition-all cursor-pointer flex justify-between items-center"
                    >
                      <div className="space-y-1 pr-4">
                        <span className="inline-block bg-slate-200 dark:bg-zinc-800 font-mono text-[10px] font-black px-1.5 py-0.5 rounded text-slate-700 dark:text-zinc-300">
                          {item.code}
                        </span>
                        <h4 className="text-xs font-bold text-slate-800 dark:text-zinc-200 group-hover:text-teal-600 dark:group-hover:text-teal-400">
                          {item.title}
                        </h4>
                      </div>
                      <span className="text-[10px] bg-emerald-100 dark:bg-emerald-950/40 text-emerald-800 dark:text-emerald-300 px-2 py-0.5 rounded-full font-bold">
                        Match: {Math.round(item.confidence * 100)}%
                      </span>
                    </div>
                  ))}
                </div>
              )}
            </div>

          </div>

          {/* RIGHT: Claim Builder Form (7 cols) */}
          <div className="lg:col-span-7">
            <ClaimBuilder
              diagnoses={diagnoses}
              setDiagnoses={setDiagnoses}
              carePathway={carePathway}
              setCarePathway={setCarePathway}
              onSubmit={handleSubmitClaim}
              isSubmitting={isSubmitting}
            />
          </div>

        </div>

        {/* Real-time Validation Report Dashboard Overlay */}
        {analysisResult && (
          <div className="border border-slate-200 dark:border-zinc-800 rounded-2xl bg-white dark:bg-zinc-900 shadow-2xl p-6 space-y-6">
            <div className="flex justify-between items-start border-b border-slate-100 dark:border-zinc-800 pb-4">
              <div>
                <span className="text-xxs font-bold uppercase tracking-widest text-teal-600 dark:text-teal-400">Real-time Claims Auditing</span>
                <h3 className="text-xl font-bold text-slate-800 dark:text-zinc-100">
                  Submission Status: <span className={analysisResult.status === 'APPROVED' ? 'text-emerald-500' : 'text-amber-500'}>{analysisResult.status}</span>
                </h3>
                <p className="text-xs text-slate-400 mt-1 font-mono">Claim Reference: {analysisResult.claimId}</p>
              </div>
              <div className="text-right">
                <span className="text-xxs font-semibold text-slate-500">Risk Assessment</span>
                <p className={`text-lg font-black ${
                  analysisResult.riskScoring.riskCategory === 'HIGH' ? 'text-red-500' : 
                  analysisResult.riskScoring.riskCategory === 'MEDIUM' ? 'text-amber-500' : 'text-emerald-500'
                }`}>{analysisResult.riskScoring.riskCategory} RISK ({analysisResult.riskScoring.overallRiskScore}/100)</p>
              </div>
            </div>

            {/* STP Compliance progress */}
            <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
              
              <div className="border border-slate-100 dark:border-zinc-800 rounded-xl p-4 space-y-3">
                <div className="flex justify-between items-center text-xs">
                  <span className="font-bold text-slate-700 dark:text-zinc-300">STP Adherence Score</span>
                  <span className="font-extrabold text-teal-600 dark:text-teal-400">{analysisResult.stpCompliance.complianceScore}%</span>
                </div>
                
                <div className="h-2 bg-slate-100 dark:bg-zinc-800 rounded-full overflow-hidden">
                  <div 
                    className="h-full bg-gradient-to-r from-teal-500 to-emerald-500 transition-all"
                    style={{ width: `${analysisResult.stpCompliance.complianceScore}%` }}
                  />
                </div>
                
                <p className="text-xxs text-slate-400 italic">
                  Matched protocol: {analysisResult.stpCompliance.protocolName}
                </p>

                {analysisResult.stpCompliance.deviations.length > 0 ? (
                  <div className="space-y-1.5 pt-2">
                    <span className="text-[10px] font-bold text-slate-400 uppercase tracking-wide">Protocol Deviations:</span>
                    {analysisResult.stpCompliance.deviations.map((dev, i) => (
                      <div key={i} className="text-xs text-amber-600 dark:text-amber-400 bg-amber-50/50 dark:bg-amber-950/10 border border-amber-100 dark:border-amber-900 p-2 rounded-lg">
                        ⚠️ <strong>[{dev.type}]</strong> {dev.message} (Penalty: -{dev.penalty})
                      </div>
                    ))}
                  </div>
                ) : (
                  <p className="text-xs text-emerald-600 dark:text-emerald-400 bg-emerald-50/50 dark:bg-emerald-950/10 border border-emerald-100 dark:border-emerald-900 p-2 rounded-lg">
                    ✅ Adheres perfectly to the Standard Treatment Protocol.
                  </p>
                )}
              </div>

              {/* Service Mapping Mismatch Warnings */}
              <div className="border border-slate-100 dark:border-zinc-800 rounded-xl p-4 space-y-3">
                <span className="text-xs font-bold text-slate-700 dark:text-zinc-300">Clinical Mismatch Audits</span>
                
                {!analysisResult.clinicalValidation.isValid ? (
                  <div className="space-y-2">
                    {analysisResult.clinicalValidation.issues.map((issue, i) => (
                      <div key={i} className="text-xs text-red-600 dark:text-red-400 bg-red-50/50 dark:bg-red-950/10 border border-red-100 dark:border-red-900 p-2.5 rounded-lg flex items-start gap-2">
                        <span>❌</span>
                        <div>
                          <strong>{issue.severity} Warning:</strong> {issue.message}
                        </div>
                      </div>
                    ))}
                  </div>
                ) : (
                  <p className="text-xs text-emerald-600 dark:text-emerald-400 bg-emerald-50/50 dark:bg-emerald-950/10 border border-emerald-100 dark:border-emerald-900 p-2.5 rounded-lg">
                    ✅ All billed medications and procedures are clinically justified by your diagnoses.
                  </p>
                )}
              </div>

            </div>

            <div className="flex justify-end pt-2">
              <button 
                onClick={() => setAnalysisResult(null)}
                className="px-4 py-2 border border-slate-200 dark:border-zinc-800 rounded-xl text-xs font-semibold hover:bg-slate-100 dark:hover:bg-zinc-800 text-slate-600 dark:text-zinc-300 transition-all cursor-pointer"
              >
                Dismiss Audit
              </button>
            </div>
          </div>
        )}

      </div>
    </div>
  );
}
